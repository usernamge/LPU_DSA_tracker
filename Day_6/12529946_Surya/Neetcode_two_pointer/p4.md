# Container With Most Water : https://neetcode.io/problems/max-water-container/question?list=neetcode150
``` Python
class Solution:
    def maxArea(self, height: List[int]) -> int:

        ansl = []
        l = len(height)

        p = 0
        q = l - 1

        while p < q:

            if height[p] < height[q]:
                ele = height[p] * (q - p)
                p += 1

            elif height[p] > height[q]:
                ele = height[q] * (q - p)
                q -= 1

            else:
                ele = height[p] * (q - p)
                p += 1      # Move one pointer

            ansl.append(ele)

        return max(ansl)
```
