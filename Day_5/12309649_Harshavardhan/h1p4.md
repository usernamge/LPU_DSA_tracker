# h1p4 Palindrome Linked List

https://leetcode.com/problems/palindrome-linked-list/submissions/2046660421/

```python

class Solution:
    def reverse(self, head):
        prev = None
        while head:
            nxt = head.next
            head.next = prev
            prev = head
            head = nxt
        return prev

    def isPalindrome(self, head):
        if not head or not head.next:
            return True

        slow = fast = head
        while fast and fast.next:
            slow = slow.next
            fast = fast.next.next

        second = self.reverse(slow)
        first = head

        while second:
            if first.val != second.val:
                return False
            first = first.next
            second = second.next

        return True
