
# • Count inversions using merge sort (dp)

```python
from typing import List

class Solution:
    def numberOfPermutations(self, n: int, requirements: List[List[int]]) -> int:
        MOD = 1_000_000_007
        
        # Map each prefix index to its required inversion count
        req_map = {end: cnt for end, cnt in requirements}
        
        # Base Case: The first element (index 0) can only ever have 0 inversions.
        # If a requirement says otherwise, it's impossible.
        if 0 in req_map and req_map[0] != 0:
            return 0
            
        max_inversions = max(cnt for _, cnt in requirements)
        
        # dp[j] stores the number of valid permutations with exactly j inversions
        dp = [0] * (max_inversions + 1)
        dp[0] = 1  # There is 1 way to arrange index 0 with 0 inversions
        
        # Process elements from index 1 up to n-1
        for i in range(1, n):
            next_dp = [0] * (max_inversions + 1)
            running_sum = 0
            
            # Use a sliding window of size (i + 1) to calculate transitions in O(1)
            for j in range(max_inversions + 1):
                running_sum += dp[j]
                if j > i:
                    running_sum -= dp[j - i - 1]
                
                running_sum %= MOD
                next_dp[j] = running_sum
            
            # If there's a strict requirement for the current prefix index `i`,
            # invalidate any state that doesn't match the required inversion count.
            if i in req_map:
                target_cnt = req_map[i]
                for j in range(max_inversions + 1):
                    if j != target_cnt:
                        next_dp[j] = 0
                        
            dp = next_dp
            
        return dp[req_map[n - 1]]
```
