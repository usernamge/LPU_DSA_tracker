Ugly Number 2: https://leetcode.com/problems/ugly-number-ii/
``` Python
class Solution:
    def nthUglyNumber(self, n: int) -> int:
        nums = [1]
        visited = {1}

        for i in range(n):
            curr = nums.pop(0)

            for f in [2,3,5]:
                nxt = curr * f
                if nxt not in visited:
                    visited.add(nxt)
                    nums.append(nxt)

            nums.sort()

        return curr
``` 
