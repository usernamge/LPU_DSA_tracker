# Koko Eating Bananas
- https://leetcode.com/problems/koko-eating-bananas/description/
``` python 
import math

class Solution:
    def minEatingSpeed(self, piles: List[int], h: int) -> int:
        
        s = sum(piles)
        l = math.ceil(s / h)
        r = max(piles)
        
        while r > l:
            s = 0
            mid = (l + r) // 2
            for p in piles:
                s += math.ceil(p / mid)
            if s > h:
                l = mid + 1
            else:
                r = mid
        return l
```
