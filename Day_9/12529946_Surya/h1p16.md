## Roman to Integer : https://leetcode.com/problems/roman-to-integer/description/?__cf_chl_f_tk=3kkpIFK1YRfo4IGkrb9jfm9da6sBVB18T8folxbdJY8-1782974029-1.0.1.1-rxl0et5TmD6H42N6fraxAcT1JGLhPUzoQESLKIH41DY

```Python
class Solution:
    def romanToInt(self, s: str) -> int:
        table = {'I':1,'V':5,'X':10,'L':50,'C':100,'D':500,'M':1000}
        sum = 0
        max_next_value = 0
        for i in reversed(s):
            value = table[i]
            if value<max_next_value:
                sum = sum - value
            else:
                sum = sum+value
                max_next_value = value
        return sum
```
