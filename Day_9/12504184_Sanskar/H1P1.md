# 412 - Fizz Buzz Number

- https://leetcode.com/problems/fizz-buzz/

```python

class Solution:
    def fizzBuzz(self, n: int) -> List[str]:
        res = ['']*n
        def isFizzBuzz(i):
            if i%3==0 and i%5==0:
                return "FizzBuzz"
            elif i%3==0:
                return "Fizz"
            elif i%5 == 0:
                return "Buzz"
            else:
                return str(i)
        
        for i in range(0,n):
            res[i] = isFizzBuzz(i+1)
    
        return res

```
