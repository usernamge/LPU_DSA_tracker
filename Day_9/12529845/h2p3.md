class Solution:
    def minEatingSpeed(self, piles: list[int], h: int) -> int:
        left = 1
        right = max(piles)
        while left < right:
            mid = (left + right) // 2
            hours = 0
            for bananas in piles:
                hours += (bananas + mid - 1) // mid
            if hours <= h:
                right = mid
            else:
                left = mid + 1
        return left
